Disabling protocol commands¶

SDK users can opt to disable specific protocol commands in a TeamSpeak server instance, so clients are unable to use these commands. The server can still issue disabled commands by calling the appropriate Server Lib functions.

Commands that can be disabled are described in the ClientCommand enum.

Note

By default all commands are allowed, unless explicitly disabled.

To disable protocol commands for all clients call

unsigned int ts3server_disableClientCommand(int clientCommand)

Prevents clients from performing certain actions. SDK only.

Use this to disable certain features for clients, e.g. deleting channels or moving clients so that the server has authority over these matters and is the only entity who can do so. To disable multiple commands, call this function once for each command you would like to disable for clients.

Return

An Error code from the Ts3ErrorType enum indicating either success or the failure reason

Parameters

  • clientCommand: the command to disable. One of the values from the ClientCommand enum

To disable multiple commands, call this function once per command to disable.

Warning

There is no way to enable commands again once disabled.

To enable all disabled commands again you will have to destroy and initialize the server lib again.

Example¶

For example, an SDK user may decide that clients should not be able to create channels on a TeamSpeak server and implement this action only on the Server side.

Any client call to ts3client_flushChannelCreation() will be rejected by the server.

#include <teamspeak/clientlib.h>
#include <teamspeak/serverlib.h>
#include <teamspeak/public_definitions.h>
#include <teamspeak/public_errors.h>
#include <stdio.h> // printf
#include <string.h> // memset

void onConnect(uint64 schid, int status, unsigned int error) {
    if (status == STATUS_CONNECTION_ESTABLISHING) {
        printf("Creating channel...\n");
        unsigned int err;
        if ((err = ts3client_setChannelVariableAsString(schid, 0, CHANNEL_NAME, "TestChannel")) != ERROR_ok) {
            printf("Failed to set channel variable: 0x%04X\n", err);
            return;
        }
        if ((err = ts3client_setChannelVariableAsInt(schid, 0, CHANNEL_FLAG_PERMANENT, 1)) != ERROR_ok) {
            printf("Failed to set channel variable: 0x%04X\n", err);
            return;
        }
        if ((err = ts3client_flushChannelCreation(schid, 0, "sdk-channel-create")) != ERROR_ok) {
            printf("Failed to set channel variable: 0x%04X\n", err);
            return;
        }
    }
}

void onError(uint64 schid, const char* message, unsigned int error, const char* returnCode, const char* extra) {
    printf("Connection %llu has error 0x%04X", schid, error);
    if (returnCode) {
        printf(" for call %s", returnCode);
    }
    printf("\n");
}

void onChannelCreated(uint64 schid, uint64 channelId, uint64 parentId, anyID invokerId, const char* invokerName, const char* invokerUid) {
    printf("Channel %llu created by %s as a child of %llu on connection %llu\n", channelId, invokerName, parentId, schid);
}

int main() {
    struct ServerLibFunctions sf;
    memset(&sf, 0, sizeof sf);
    struct ClientUIFunctions cf;
    memset(&cf, 0, sizeof cf);
    cf.onConnectStatusChangeEvent = onConnect;
    cf.onServerErrorEvent = onError;
    cf.onNewChannelCreatedEvent = onChannelCreated;
    unsigned int err = ts3client_initClientLib(&cf, NULL, LogType_NONE, "", "");
    if (err != ERROR_ok) {
        printf("Failed to init lib: 0x%04X\n", err);
        return 1;
    }
    if ((err = ts3server_initServerLib(&sf, LogType_NONE, "")) != ERROR_ok) {
        printf("Failed to init server lib: 0x%04X\n", err);
        return 2;
    }
    ts3server_disableClientCommand(CLIENT_COMMAND_flushChannelCreation);
    uint64 serverId = 0;
    if ((err = ts3server_createVirtualServer(9987, "127.0.0.1", "TestServer", NULL, 4, &serverId)) != ERROR_ok) {
        printf("Failed to create server: 0x%04X\n", err);
        return 2;
    }

    uint64 clientTab = 0;
    ts3client_spawnNewServerConnectionHandler(0, &clientTab);

    char* ident = NULL;
    ts3client_createIdentity(&ident);
    if ((err = ts3client_startConnection(clientTab, ident, "127.0.0.1", 9987, "TestUser", NULL, "", "")) != ERROR_ok) {
        printf("Failed to connect: 0x%04X\n", err);
        return 1;
    }

    printf("Press Enter to terminate\n");
    getchar();

    printf("Terminating...\n");
    ts3client_stopConnection(clientTab, "bye");
    ts3server_stopVirtualServer(serverId);
    ts3client_destroyServerConnectionHandler(clientTab);
    ts3client_destroyClientLib();
    ts3server_destroyServerLib();

    return 0;
}
